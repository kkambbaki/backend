9 BOT 토큰을 사용한 인증 클래스

16 def authenticate(self, request): 

17 """ 

18 HTTP 헤더에서 BOT 토큰을 추출하고 검증하여 사용자를 인증합니다. 

19 

20 Args: 

21 request: DRF Request 객체 

22 

23 Returns: 

24 tuple: (user, token) 또는 None 

25 """ 

26 # 헤더에서 토큰 추출 

27 bot_token = self.get_token_from_request(request) 

28 

29 if bot_token is None: 

30 return None 

31 

32 return self.authenticate_credentials(bot_token)

34 def get_token_from_request(self, request): 

35 """ 

36 Request 객체에서 BOT 토큰을 추출합니다. 

37 X-BOT-TOKEN 헤더 또는 쿼리 파라미터에서 토큰을 찾습니다. 

38 

39 Args: 

40 request: DRF Request 객체 

41 

42 Returns: 

43 str: 토큰 문자열 또는 None 

44 """ 

45 # HTTP 헤더에서 토큰 확인 

46 token = request.META.get(self.header_name) 

47 

48 if token: 

49 return token 

50 

51 # 쿼리 파라미터에서 토큰 확인 (fallback) 

52 token = request.query_params.get("BOT_TOKEN") 

53 

54 return token

56 def authenticate_credentials(self, token): 

57 """ 

58 토큰을 검증하고 해당 레포트의 사용자를 반환합니다. 

59 

60 Args: 

61 token: BOT 토큰 문자열 

62 

63 Returns: 

64 tuple: (user, token) 

65 

66 Raises: 

67 AuthenticationFailed: 토큰이 유효하지 않은 경우 

68 """ 

69 # verify_token 메서드 사용하여 사용자 정보와 유효성 확인 

70 user, is_valid = BotToken.verify_token(token) 

71 

72 if not is_valid or user is None: 

73 raise exceptions.AuthenticationFailed("Invalid BOT token") 

74 

75 if not user.is_active: 

76 raise exceptions.AuthenticationFailed("User inactive or deleted") 

77 

78 return (user, token)
