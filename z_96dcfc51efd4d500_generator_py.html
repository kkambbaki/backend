19class ReportAdviceGenerator(ABC): 

20 def __init__( 

21 self, 

22 provider_name: Optional[str] = None, 

23 temperature: float = 0.7, 

24 max_retries: int = 5, 

25 ): 

26 """ 

27 Args: 

28 provider_name: LLM provider name (ollama, openai, gemini) 

29 temperature: Sampling temperature 

30 max_retries: Maximum number of retries on failure 

31 """ 

32 self.provider_name = provider_name or LLM_PROVIDER 

33 self.temperature = temperature 

34 self.max_retries = max_retries 

35 

36 # LLM Chat 모델 초기화 

37 self.llm = get_chat_model(self.provider_name, self.temperature) 

38 

39 # 프롬프트 및 파서 초기화 

40 self.prompt = self.get_prompt() 

41 self.parser = JsonOutputParser(pydantic_object=self.get_response_model()) 

42 

43 # 체인 생성 

44 self.chain = self.prompt | self.llm | self.parser

46 def generate_advice(self, report_data: dict) -> list: 

47 """레포트 데이터를 기반으로 조언 생성 (Template Method) 

48 

49 Args: 

50 report_data: 레포트 정보를 담은 딕셔너리 

51 

52 Returns: 

53 생성된 조언 딕셔너리 리스트, 각 딕셔너리는 'title'과 'description' 키를 포함 

54 """ 

55 for attempt in range(self.max_retries): 

56 try: 

57 payload = self._prepare_payload(report_data) 

58 

59 # LLM 체인 실행 

60 result = self.chain.invoke(payload) 

61 

62 # 응답 파싱 및 검증 

63 response = GameReportAdviceResponse(**result) 

64 

65 return [{"title": item.title, "description": item.description} for item in response.analysis] 

66 

67 except ValidationError as e: 

68 if attempt < self.max_retries - 1: 

69 time.sleep(1) 

70 continue 

71 raise ValueError(f"LLM 응답 검증 실패: {e}") 

72 

73 except json.JSONDecodeError as e: 

74 if attempt < self.max_retries - 1: 

75 time.sleep(1) 

76 continue 

77 raise ValueError(f"LLM 응답 JSON 파싱 실패: {e}") 

78 

79 except Exception as e: 

80 if attempt < self.max_retries - 1: 

81 time.sleep(1) 

82 continue 

83 raise ValueError(f"조언 생성 중 오류 발생: {e}") 

84 

85 raise ValueError("최대 재시도 횟수 초과")
