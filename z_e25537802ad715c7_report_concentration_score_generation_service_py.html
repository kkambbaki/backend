27class ReportConcentrationScoreGenerationService: 

28 """ 

29 레포트 집중력 점수 생성 서비스 

30 

31 점수는 0-100점 사이이며, 다음 요소를 고려합니다: 

32 1. 전체 성공률 (40%) 

33 2. 최대 라운드 도달률 (30%) 

34 3. 플레이 일관성 (30%) 

35 4. 최근 개선도 (10%) - 보너스 점수 

36 총 110% 구성, 100점 초과 시 100점으로 커팅 

37 """

39 @staticmethod 

40 def calculate_concentration_score(report: Report) -> int: 

41 """ 

42 레포트의 집중력 점수 계산 

43 

44 Args: 

45 report: Report 객체 

46 

47 Returns: 

48 int: 0-100 사이의 집중력 점수 

49 """ 

50 game_reports = GameReport.objects.by_report(report).select_related("game") 

51 

52 if not game_reports.exists(): 

53 logger.info(f"No game reports found for report {report.id}. Returning score 0.") 

54 return 0 

55 

56 total_score = 0.0 

57 game_count = 0 

58 

59 for game_report in game_reports: 

60 if game_report.total_plays_count == 0: 

61 continue 

62 

63 game_score = ReportConcentrationScoreGenerationService._calculate_game_score(game_report) 

64 total_score += game_score 

65 game_count += 1 

66 

67 if game_count == 0: 

68 return 0 

69 

70 # 평균 점수 계산 

71 average_score = total_score / game_count 

72 

73 # 0-100 범위로 제한 (110% 구성이므로 100 초과 가능, 커팅 필요) 

74 final_score = max(0, min(100, int(round(average_score)))) 

75 

76 logger.info(f"Calculated concentration score for report {report.id}: {final_score}") 

77 

78 return final_score

81 def _calculate_game_score(game_report: GameReport) -> float: 

82 """ 

83 개별 게임의 집중력 점수 계산 

84 

85 Args: 

86 game_report: GameReport 객체 

87 

88 Returns: 

89 float: 0-110 사이의 점수 (100 초과 가능) 

90 """ 

91 # 1. 성공률 점수 (40%) 

92 success_rate_score = ReportConcentrationScoreGenerationService._calculate_success_rate_score(game_report) 

93 

94 # 2. 최대 라운드 도달 점수 (30%) 

95 max_round_score = ReportConcentrationScoreGenerationService._calculate_max_round_score(game_report) 

96 

97 # 3. 일관성 점수 (30%) 

98 consistency_score = ReportConcentrationScoreGenerationService._calculate_consistency_score(game_report) 

99 

100 # 4. 최근 개선도 점수 (10%) - 보너스 

101 improvement_score = ReportConcentrationScoreGenerationService._calculate_improvement_score(game_report) 

102 

103 # 가중치 적용 (총 110% 가능) 

104 total_score = ( 

105 success_rate_score * WEIGHT_SUCCESS_RATE 

106 + max_round_score * WEIGHT_MAX_ROUND_ACHIEVEMENT 

107 + consistency_score * WEIGHT_CONSISTENCY 

108 + improvement_score * WEIGHT_RECENT_IMPROVEMENT 

109 ) 

110 

111 logger.debug( 

112 f"Game {game_report.game.code} scores - " 

113 f"Success: {success_rate_score:.1f}, " 

114 f"MaxRound: {max_round_score:.1f}, " 

115 f"Consistency: {consistency_score:.1f}, " 

116 f"Improvement: {improvement_score:.1f}, " 

117 f"Total: {total_score:.1f}" 

118 ) 

119 

120 return total_score

122 @staticmethod 

123 def _calculate_success_rate_score(game_report: GameReport) -> float: 

124 """ 

125 성공률 기반 점수 계산 

126 

127 Args: 

128 game_report: GameReport 객체 

129 

130 Returns: 

131 float: 0-100 사이의 점수 

132 """ 

133 if game_report.total_play_actions_count == 0: 

134 return 0.0 

135 

136 success_rate = (game_report.total_success_count / game_report.total_play_actions_count) * 100 

137 

138 # 성공률을 그대로 점수로 사용 

139 return min(100.0, success_rate)

141 @staticmethod 

142 def _calculate_max_round_score(game_report: GameReport) -> float: 

143 """ 

144 최대 라운드 도달률 기반 점수 계산 

145 

146 Args: 

147 game_report: GameReport 객체 

148 

149 Returns: 

150 float: 0-100 사이의 점수 

151 """ 

152 if game_report.total_plays_count == 0: 

153 return 0.0 

154 

155 # 최대 라운드 도달 비율 

156 max_round_achievement_rate = (game_report.max_rounds_count / game_report.total_plays_count) * 100 

157 

158 # 평균 도달 라운드 비율 

159 avg_rounds_count = game_report.get_avg_rounds_count() 

160 if avg_rounds_count is None: 

161 avg_round_rate = 0.0 

162 else: 

163 avg_round_rate = (avg_rounds_count / game_report.game.max_round) * 100 

164 

165 # 두 지표의 평균 사용 (최대 도달률 60%, 평균 도달률 40%) 

166 score = max_round_achievement_rate * 0.6 + avg_round_rate * 0.4 

167 

168 return min(100.0, score)

170 @staticmethod 

171 def _calculate_consistency_score(game_report: GameReport) -> float: 

172 """ 

173 플레이 일관성 점수 계산 (오답률의 역수 기반) 

174 

175 Args: 

176 game_report: GameReport 객체 

177 

178 Returns: 

179 float: 0-100 사이의 점수 

180 """ 

181 wrong_rate = game_report.get_wrong_rate() 

182 

183 if wrong_rate is None: 

184 return 0.0 

185 

186 # 오답률이 낮을수록 높은 점수 

187 # 오답률 0% = 100점, 오답률 100% = 0점 

188 consistency_score = 100.0 - wrong_rate 

189 

190 return max(0.0, consistency_score)

192 @staticmethod 

193 def _calculate_improvement_score(game_report: GameReport) -> float: 

194 """ 

195 최근 게임 결과의 개선도 점수 계산 (보너스 점수) 

196 

197 최근 2개 게임의 평균 성공률과 나머지 전체 게임의 평균 성공률을 비교 

198 

199 Args: 

200 game_report: GameReport 객체 

201 

202 Returns: 

203 float: 0-100 사이의 점수 

204 """ 

205 

206 all_results_queryset = game_report.get_game_results() 

207 total_count = all_results_queryset.count() 

208 

209 if total_count < 2: 

210 # 비교할 데이터가 부족하면 기본 점수 반환 (50점) 

211 return 50.0 

212 

213 # ORM을 사용하여 성공률 계산 

214 # success_rate = success_count / (success_count + wrong_count) * 100 

215 results_with_rate = all_results_queryset.annotate( 

216 total_actions=F("success_count") + F("wrong_count"), 

217 success_rate=Case( 

218 When( 

219 total_actions__gt=0, 

220 then=Cast(F("success_count"), FloatField()) / Cast(F("total_actions"), FloatField()) * 100, 

221 ), 

222 default=0.0, 

223 output_field=FloatField(), 

224 ), 

225 ).filter(total_actions__gt=0) 

226 

227 valid_count = results_with_rate.count() 

228 

229 if valid_count < 2: 

230 return 50.0 

231 

232 # 최근 2개 게임의 평균 성공률 

233 recent_results = results_with_rate[:RECENT_GAMES_COUNT] 

234 recent_avg = recent_results.aggregate(avg_rate=Avg("success_rate"))["avg_rate"] or 0.0 

235 

236 # 나머지 전체 게임의 평균 성공률 

237 if valid_count > RECENT_GAMES_COUNT: 

238 older_results = results_with_rate[RECENT_GAMES_COUNT:] 

239 older_avg = older_results.aggregate(avg_rate=Avg("success_rate"))["avg_rate"] or 0.0 

240 else: 

241 # 나머지가 없으면 최근 평균과 동일하게 설정 (개선도 0) 

242 older_avg = recent_avg 

243 

244 # 개선도 계산 (-100 ~ +100) 

245 improvement = recent_avg - older_avg 

246 

247 # 0-100 스케일로 변환 

248 # 개선도가 +20% 이상이면 100점 

249 # 개선도가 0%이면 50점 

250 # 개선도가 -20% 이하이면 0점 

251 improvement_score = 50 + (improvement * 2.5) 

252 

253 return max(0.0, min(100.0, improvement_score))
